    "PayoffFn = Callable[[tuple[bool, ...]], float]\n",
    "\n",
    "\n",
    "def interaction_payoff(present: tuple[bool, ...], num_players: int) -> float:\n",
    "    coefficients = tuple(float(index + 1) for index in range(num_players))\n",
    "    linear = sum(\n",
    "        coefficient\n",
    "        for coefficient, is_present in zip(coefficients, present, strict=True)\n",
    "        if is_present\n",
    "    )\n",
    "    return linear + 0.2 * (linear * linear) / num_players\n",
    "\n",
    "\n",
    "def exact_reference(num_players: int) -> tuple[list[float], float, tuple[CoalitionMask, ...]]:\n",
    "    player_set = PlayerSet(player_ids=tuple(f\"p{index}\" for index in range(num_players)))\n",
    "    payoff_fn: PayoffFn = lambda present: interaction_payoff(present, num_players)\n",
    "    all_masks = tuple(\n",
    "        CoalitionMask.from_sequence(mask)\n",
    "        for mask in product([False, True], repeat=num_players)\n",
    "        if not all(mask)\n",
    "    )\n",
    "    all_payoffs = [payoff_fn(mask.present) for mask in all_masks]\n",
    "    exact_values = ExactEstimator().estimate_attributions(\n",
    "        all_masks,\n",
    "        all_payoffs,\n",
    "        ShapleyAggregator(),\n",
    "    )\n",
    "    scale = sum(abs(value) for value in exact_values)\n",
    "    return exact_values, scale, all_masks\n",
    "\n",
    "\n",
    "def mc_fraction_for_cap(num_players: int, target_cap: int) -> float:\n",

    "\n",

    "## Why Monte Carlo shows high error but tiny variance\n",
    "\n",
    "**Business:** A narrow error band is **not** proof of accuracy — MC can be confidently wrong.\n",
    "\n",
    "**Technical:**\n",
    "\n",
    "1. **`include_minimal_masks=True`** emits the same empty/singleton prefix on every seed. When the cap is ≤ `n + 1`, the sample is **deterministic**.\n",
    "2. **`ShapleyAggregator.aggregate_from_marginals`** skips Shapley terms unless both `S` and `S ∪ {i}` were sampled — most terms never enter the sum."

    "def shapley_marginal_coverage(\n",
    "    masks: list[CoalitionMask],\n",
    "    num_players: int,\n",
    ") -> tuple[int, int]:\n",
    "    present_lookup = {present_to_mask_int(mask.present) for mask in masks}\n",
    "    resolved = 0\n",
    "    possible = 0\n",
    "    for player_index in range(num_players):\n",
    "        player_bit = 1 << player_index\n",
    "        for mask in masks:\n",
    "            mask_int = present_to_mask_int(mask.present)\n",
    "            if mask_int & player_bit:\n",
    "                continue\n",
    "            possible += 1\n",
    "            if (mask_int | player_bit) in present_lookup:\n",
    "                resolved += 1\n",
    "    return resolved, possible\n",
